```sh filename="install"
curl -fsSL https://bun.sh/install | bash
```

```ts filename="index.ts"
console.log("Hello via Bun!");
```

- 開発サーバー用にscriptsを追加する

```json {5-8} filename="package.json" 
{
  "name": "hono_tutrial",
  "module": "index.ts",
  "type": "module",
  "scripts": {
    "start": "bun index.ts",
    "dev": "bun --watch index.ts"
  },
  "devDependencies": {
    "@types/bun": "latest"
  },
  "peerDependencies": {
    "typescript": "^5.0.0"
  }
}
```

```sh filename="実行"
bun dev
```

## HTTPサーバーを作る
[bun docs > http](https://bun.sh/docs/api/http)
```ts filename="index.ts"
Bun.serve({
  fetch(req) {
    return new Response("Bun!");
  },
});

console.log("Server running");
```
